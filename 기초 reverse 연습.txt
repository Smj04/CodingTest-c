#include <stdio.h>
int main() {

	
	int i = 0;
for(i = 0; i<10; i++){
	printf("%d ",i);
}
// i++이니까 결국에 1이 증가하고 끝남.
//그래서 j문에 -1을 해줘야됨
//배열이면 [-1]

printf("\n");
//여기서 -1은 출력이 어디까지 나올 값
for (int j = i-1; j >= 0; j--) {
	printf("%d ", j);
}
//i-1을 해줬기 때문에 j>=이퀄까지 들어감

	return 0;
}
