#include <iostream>
using namespace std;

int main() {
	int size = 0;
	
	cout << "방 갯수 : ";
	cin >> size;
	double arr[1000];
	double max = arr[0];
	double tot = 0;
	int i = 0;
	for (i = 0; i < size; i++) {
		cin >> arr[i];
		
		if (max < arr[i]){
			max = arr[i];
		}
	}
	for (i = 0; i < size; i++) {
		arr[i] = arr[i] / max * 100;
		tot += arr[i];
	}
	double avg = tot / size;
	cout << (avg);


	return 0;
}
