#include <stdio.h>
int main() {
	int size; //입력 받을 크기
	printf("생성할 방 갯수 : ");
	scanf("%d", &size);
	int* num = malloc(sizeof(int) * size); //입력 받은 크기만큼 
	//동적할당하여 배열 방생성
	
	for (int i = 0; i < size; i++) {
		num[i] = i; //방을 i만큼 할당
	}
	for (int i = 0; i < size; i++) {
		printf("%d\n", num[i]); // 출력문
	}
	return 0;
}
